165 // .........................................................
166 //
167 // Ejemplo de Beacon (31 bytes)
168 //
169 // https://os.mbed.com/blog/entry/BLE-Beacons-URIBeacon-AltBeacons-iBeacon/
170 //
171 // The iBeacon Prefix contains the hex data : 0x0201061AFF004C0215. This breaks down as follows:
172 //
173 // 0x020106 defines the advertising packet as BLE General Discoverable and BR/EDR high-speed incompatible.
174 // Effectively it says this is only broadcasting, not connecting.
175 //
176 // 0x1AFF says the following data is 26 bytes long and is Manufacturer Specific Data.
177 //
178 // 0x004C is Apple’s Bluetooth Sig ID and is the part of this spec that makes it Apple-dependent.
179 //
180 // 0x02 is a secondary ID that denotes a proximity beacon, which is used by all iBeacons.
181 //
182 // 0x15 defines the remaining length to be 21 bytes (16+2+2+1).
183 //
184 // Por ejemmplo:
185 //
186 // 1. prefijo: 9bytes
187 // 0x02, 0x01, 0x06, // advFlags 3bytes
188 // 0x1a, 0xff, // advHeader 2 (0x1a = 26 = 25(lenght de 0x4c a 0xca)+1) 0xFF -> BLE_GAP_AD_TYPE_MANUFACTURER_SPECIFIC_DATA
189 // 0x4c, 0x00, // companyID 2bytes
190 // 0x02, // ibeacon type 1 byte
191 // 0x15, // ibeacon length 1 byte (dec=21 lo que va a continuación: desde la 'f' hasta 0x01)
192 //
193 // 2. uuid: 16bytes
194 // 'f', 'i', 's', 't', 'r', 'o', 'f', 'i', 's', 't', 'r', 'o', 0xa7, 0x10, 0x96, 0xe0
195 //
196 // 2 major: 2bytes
197 // 0x04, 0xd2,
198 //
199 // minor: 2bytes
200 // 0x10, 0xe1,
201 //
202 // 0xca, // tx power : 1bytes
203 //
204 // 0x01, // este es el byte 31 = BLE_GAP_ADV_SET_DATA_SIZE_MAX, parece que sobra
205 //
206 // .........................................................
207 // Para enviar como carga libre los últimos 21 bytes de un iBeacon (lo que normalmente sería uuid-16 major-2 minor-2 txPower-1)
208 // .........................................................
